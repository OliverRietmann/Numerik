import numpy as np
import matplotlib.pyplot as plt

n = 4
x = np.linspace(-2.5, 2.5, n)
y = 2.0 * x + 3.0 + 0.5 * np.random.rand(n)

A = np.column_stack((np.ones_like(x), x))
AT = np.transpose(A)

b, m = np.linalg.solve(AT @ A, np.dot(AT, y))

plt.figure()
plt.plot(x, y, 'bo')
plt.plot(x, m * x + b, 'r-')
plt.show()
